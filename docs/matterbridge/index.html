Matterbridge

docker-compose.yaml

Nosso dockerfile

version: '3.7'
services:
  matterbridge:
    container_name: matterbridge_bolhatech
    image: 42wim/matterbridge:1.25
    restart: unless-stopped
    volumes:
    - ./matterbridge.toml:/etc/matterbridge/matterbridge.toml:ro
